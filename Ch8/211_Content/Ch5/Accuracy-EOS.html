def kappa(w):
        return 0.37464+1.54266*w-0.26992*w**2
def alpha(Tr,w):    
    return (1+kappa(w)*(1-np.sqrt(Tr)))**2

def a(T,Tc,Pc,w):
    Tr=T/Tc
    return 0.45724*(R*Tc)**2/Pc*alpha(Tr,w)

def b(Tc,Pc):
    return 0.07780*R*Tc/(Pc)

def amix(x1):
    x2=(1-x1)
    return a1*x1**2+a2*x2**2+x1*x2*np.sqrt(a1*a2)*(1-kij)
    
def bmix(x1):
    x2=(1-x1)
    return b1*x1+b2*x2

x1=np.linspace(0,1)
amix_=amix(x1)
bmix_=bmix(x1)

plt.plot(x1,amix_)
plt.plot(x1,bmix_)

def Zcoeff(T,P,a,b):
    A=a*P/(R*T)**2
    B=b*P/(R*T)
    
    return [1,
            -1+B,
            A-3*B**2-2*B,
            -A*B+B**2+B**3]
